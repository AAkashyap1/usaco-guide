## Solution w/ Graphs

What if there were up to $10^5$ cows? Then neither of the solutions provided in
the analysis would be fast enough.

Say that we draw an edge between two cows if they must be adjacent. Then if a
solution exists, the graph must be a union of paths.

(insert diagram)

We iterate over the cows in increasing lexicographical order. If a cow is not
yet part of the answer and it is the endpoint of a path, then we add the entire
path containing the cow to the answer.

```cpp
#include <bits/stdc++.h>
using namespace std;

// lambda expression - see https://usaco.guide/general/lambda
vector<string> cows = []() {
	vector<string> tmp{
		"Bessie", "Buttercup", "Belinda", "Beatrice",
		"Bella", "Blue", "Betsy", "Sue"
	};
	sort(begin(tmp), end(tmp));  // sort names lexicographically
	return tmp;
}();

// Returns the index of the cow
int index(string cow) {
	return find(begin(cows), end(cows), cow) - begin(cows);
}
vector<int> adj[8];

int main() {
	// fastio
	cin.tie(0);
	ios::sync_with_stdio(0);
	freopen("lineup.in", "r", stdin);
	freopen("lineup.out", "w", stdout);

	int N;
	cin >> N;
	for (int i = 0; i < N; i++) {
		// Input Cow Values (excluding useless text)
		string cow1, cow2, trash;
		cin >> cow1 >> trash >> trash >> trash >> trash >> cow2;

		// convert names to indices
		int c1 = index(cow1);
		int c2 = index(cow2);

		// Add to Adjecency List
		adj[c1].push_back(c2);
		adj[c2].push_back(c1);
	}

	vector<int> seq;  // sequence of cow indices
	vector<bool> done(8);
	for (int i = 0; i < 8; i++) {
		if (!done[i] && adj[i].size() <= 1) {
			// add path to sequence
			int cur = i;
			while (adj[cur].size() == 1) {
				seq.push_back(cur);
				done[cur] = 1;
				int nex = adj[cur][0];
				adj[nex].erase(find(begin(adj[nex]), end(adj[nex]), cur));
				cur = nex;
			}
			seq.push_back(cur);
			done[cur] = 1;
		}
	}

	// Print Output
	for (int i = 0; i < seq.size(); i++) {
		int t = seq[i];
		cout << cows[t] << '\n';
	}
}
```

## Brute Force with Itertools

<Warning>
Due to the brute force nature of this solution, it is very slow and is not the optimal solution.
</Warning>
We can easily iterate through every possible permutation of cows in lexographic order using Itertools. Then we can check for the first permutation that meets the adjacency constraints and this will be our answer. 

**Time Complexity:** $\mathcal{O}(NC \cdot C!)$, where $C$ represents the number of cows 
```py
NUM_COWS = 8
from itertools import permutations

data = open('lineup.in').read().strip().split('\n')
n = int(data[0])

cows = sorted(['Bessie', 'Buttercup', 'Belinda', 'Beatrice', 
			      'Bella', 'Blue', 'Betsy', 'Sue'])
adjacents = []

for i in range(n):
	temp = data[i + 1].split(' ')
	adjacents.append([temp[0], temp[-1]])

ans = ''

orderings = [list(permutation) for permutation in list(permutations(cows))]

for i in orderings:
  count = 0
  for j in adjacents:
    if abs(i.index(j[0]) - i.index(j[1])) == 1:
      count += 1
  if count == n:
    for k in range(NUM_COWS):
      if k != 7:
          ans += (i[k] + '\n')
      else:
          ans += i[k]
    break

print(ans, file=open('lineup.out', 'w'))
```
